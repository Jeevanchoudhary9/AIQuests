            {`useEffect(() => {
  const abortController = new AbortController();

  const fetchData = async () => {
    try {
      const result = await fetch(url, {
        signal: abortController.signal
      });
      const data = await result.json();
      setData(data);
    } catch (error) {
      if (error.name === 'AbortError') {
        return;
      }
      setError(error);
    }
  };

  fetchData();

  return () => {
    abortController.abort();
  };
}, [url]);`}
